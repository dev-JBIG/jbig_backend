    # 데이터 불러오기
import pandas as pd
df = pd.read_csv('데이터이름.csv')

# 전체보기
print(df)

# 일부만 보기
# 데이터 첫 5개의 행(head)과 마지막 5개의 행(tail)을 출력하는 함수입니다.
df.head()
df.tail()

# 데이터 개수 확인
len(df)

# 데이터 정보 확인하기(각 열의 데이터 타입, 결측 값을 제외한 데이터 값 개수와 메모리 사용량 등)
df.info()

# 데이터 행렬 정보 확인하기 (행, 렬)
df.shape

# 각 열의 데이터 타입을 출력하는 함수
df.dtypes

# 각 칼럼명을 확인하는 함수
df.columns

# 각 열마다 결측 값을 제외한 나머지 데이터의 기본 통계량을 출력(max, min, mean, std 등)
df.describe()

# 에 있는 유일한 값을 넘파이 array 형태로 출력하는 함수
df.열이름.unique()
df['열이름'].unique()

# 열에 있는 유일한 값과 각 값 별로 개수를 판다스 series 형태로 출력하는 함수
df.열이름.value_counts()
df['열이름'].value_counts()
